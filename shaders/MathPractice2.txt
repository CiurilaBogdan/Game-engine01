#version 330 core
layout (location = 0) in vec4 aPos;
layout (location = 1) in vec2 aTexCoord;

uniform mat4 mvp;

out vec2 TexCoord;

void main()
{
	gl_Position = aPos * mvp;
	TexCoord = vec2(aTexCoord.x, aTexCoord.y);
}



~
#version 330 core
out vec4 FragColor;

in vec2 TexCoord;

// texture samplers
uniform sampler2D texture1;

void main()
{
	FragColor = texture(texture1, TexCoord);
}
